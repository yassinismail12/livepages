<h1>Connect Your Facebook Page</h1>
<button onclick="loginWithFacebook()">Login with Facebook</button>
<pre id="output"></pre>

<h1>Connect Your Instagram Account</h1>
<button onclick="loginWithInstagram()">Login with Instagram</button>
<pre id="ig-output"></pre>

<script>
  // Facebook login
  function loginWithFacebook() {
    FB.login(function(response) {
      if (response.authResponse) {
        FB.api('/me/accounts', function(pages) {
          document.getElementById('output').textContent =
            JSON.stringify(pages, null, 2);
        });
      } else {
        alert('User cancelled login or did not fully authorize.');
      }
    }, {scope: 'pages_show_list,pages_messaging'});
  }

  // Instagram login
  function loginWithInstagram() {
    FB.login(function(response) {
      if (response.authResponse) {
        // Replace {IG-BUSINESS-ACCOUNT-ID} with the actual connected IG account
        FB.api('/me?fields=instagram_business_account{name,username,id}', function(account) {
          document.getElementById('ig-output').textContent =
            JSON.stringify(account.instagram_business_account, null, 2);
        });
      } else {
        alert('User cancelled login or did not fully authorize.');
      }
    }, {scope: 'instagram_basic,instagram_manage_messages'});
  }

  window.fbAsyncInit = function() {
    FB.init({
      appId: '1137959874942361',  // Your App ID
      xfbml: true,
      version: 'v19.0'
    });
  };

  (function(d, s, id){
    var js, fjs = d.getElementsByTagName(s)[0];
    if (d.getElementById(id)) return;
    js = d.createElement(s); js.id = id;
    js.src = "https://connect.facebook.net/en_US/sdk.js";
    fjs.parentNode.insertBefore(js, fjs);
  }(document, 'script', 'facebook-jssdk'));
</script>
